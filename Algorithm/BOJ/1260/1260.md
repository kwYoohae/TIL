### 문제 풀이
이번 문제는 `BFS`와 `DFS`를 구현할 수 있는지에 대한 문제인것 같았습니다.

문제 조건중에 **정점이 여러 개인 경우에는 정점 번호가 작은 것을 먼저 방문하고**라는 말이 있어서 참고하여

`Priority Queue`를 이용해서 작성을 하였습니다. 

문제를 풀면서 `priority queue`의 사용법에 대해서 친숙해졌고 또한, `vecotor`의 초기화 방법을 배웠습니다.

처음 풀면서 당연하게 양방향이 아닌 단방향 그래프라고 생각했었는데 예제를 보니 양방향 그래프인 것을 확인하고 코드를 수정했습니다.

또한, 오랜만의 C++ 코드라서 Path를 Parameter로 넘겼을 때 당연히 저장이 되는 줄 알았는데 포인터를 쓰지 않아 그것 또한 수정하였습니다.

### 정답 코드
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

bool visited[1001];

void DFS(vector< priority_queue<int, vector<int>, greater<int>> > graph, vector<int> *path, int number)
{
    if(!visited[number]) // 방문하지 않은 정점일 경우만
    {
        path->push_back(number); // paht에 정점 추가
        visited[number] = true; // 방문 확인
        while(!graph[number].empty()) // while문을 통해서 모든 Edge 탐색
        {
            int front = graph[number].top();
            graph[number].pop();
            DFS(graph, path, front); // 재귀적으로 DFS 방문
        }
    }
}

void BFS(vector< priority_queue<int, vector<int>, greater<int>> > graph, vector<int> *path, int number)
{
    queue<int> q; // queue를 통해서 방문
    path->push_back(number); // path에 시작 정점 추가
    q.push(number); // queue에 시작 정점 추가 
    visited[number] = true; // 시작 정점 방문
    while(!q.empty()) // queue가 비어 있을 때 까지
    {
        int front = q.front(); // front의 값을 보고 해당 연결된 모든 Edge를 가져온다.
        q.pop();
        while(!graph[front].empty()) // 모든 Edge 탐방
        {
            if(!visited[graph[front].top()]) // 만약 방문 안한 Edge라면 방문 처리 후 queue와 path에 추가
            {
                visited[graph[front].top()] = true;
                q.push(graph[front].top());
                path->push_back(graph[front].top());
            }
            graph[front].pop();
        }
    }
}

int main()
{
    int N, M, V;
    pair<int, int> edge;
    vector< priority_queue<int, vector<int>, greater<int>>> graph(1001); // graph를 1001사이즈로 초기ㅗ하
    vector<int> Path;
    fill(visited, visited+1001, false); // visited를 모두 flase로 초기화
    cin >> N >> M >> V;
    
    for(int i = 0 ; i < M ; i++) // 각 간선은 양방향 그래프이다. 
    {
        cin >> edge.first >> edge.second;
        graph[edge.first].push(edge.second);
        graph[edge.second].push(edge.first);
    }
    DFS(graph, &Path, V);
    for(int i = 0 ; i < Path.size(); i++)
    {
        cout << Path[i] << " ";
    }
    Path.clear(); // path 초기화

    fill(visited, visited+1001, false); // visited를 다시 false로 초기화
    BFS(graph, &Path,V);
    cout << endl;
    
    for(int i = 0 ; i < Path.size(); i++)
    {
        cout << Path[i] << " ";
    }
    
}
```
